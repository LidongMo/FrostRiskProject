Univariate linear models

#set groups (biomes) to loop through 
biomes          = unique(climate_tab$biome)


# Loop to get standardized coefficients from univariate linear model
####################################################################

#Create desired dataframes
coeffDataFrameLSF           = as.data.frame(matrix(ncol = length(biomes), nrow = length(VariablesVector)))
pValDataFrameLSF            = as.data.frame(matrix(ncol = length(biomes), nrow = length(VariablesVector)))
colnames(coeffDataFrameLSF) = biomes
colnames(pValDataFrameLSF)  = biomes
rownames(coeffDataFrameLSF) = VariablesVector
rownames(pValDataFrameLSF)  = VariablesVector

#start counter
i=1

#Loop through biomes and variables
for (biome in biomes){
  tab.sub          = climate_tab[climate_tab$biome==biome,]
  coeffVect        = vector()
  pValVect         = vector()
  for (var in VariablesVector)
  {
    # train single linear model for the top ten variables
    lmFormula    = as.formula(paste("standard(sqrt(M_DegreeZeroToFrostZero)) ~ standard(",var,")", sep=""))
    linearModel  = summary(lm(lmFormula ,data=tab.sub))
    coeffSingle  = linearModel$coefficients[2]
    pValSingle   = linearModel$coefficients[2,4]
    # append this to the vectors
    coeffVect    = append(coeffVect,coeffSingle)
    pValVect     = append(pValVect,pValSingle)
  }
  # add the coeeficients and p-values to data frames
  coeffDataFrameLSF[,i] = coeffVect
  pValDataFrameLSF[,i]  = pValVect
  #monitor progress
  #print(paste(i/length(biomes)*100,"% Done"))
  i=i+1
}

# transform to matrix (required for corrplot package) and scale values from -0.75 to 0.75
coeffDataFrameLSF     = as.matrix(coeffDataFrameLSF*1.35)
pValDataFrameLSF      = as.matrix(pValDataFrameLSF)
