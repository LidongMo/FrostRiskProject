Load packages and functions

library(tidyverse)
library(data.table)
library(sp)
library(raster)
require(maptools)



Set directories and load data

data.dir = "/Users/consti/Desktop/PhD/Publication_material/24) Frost risk/R_data/GFBi_tables_for_random_forest/Leafout_tables"
knitr::opts_knit$set(root.dir = data.dir)
tbl_fread = list.files(pattern = "*.csv") %>% map_df(~fread(., stringsAsFactors = FALSE))
coordinates(tbl_fread) = c("x", "y")

#directories
raster.frost.dir = "/Users/consti/Desktop/PhD/Publication_material/24) Frost risk/R_data/CompleteRasters"
oceans.dir       = "/Users/consti/Desktop/PhD/Data/Global_layers/Environmental_layers/oceans"
land.dir         = "/Users/consti/Desktop/PhD/Data/Global_layers/Environmental_layers/land"
output.dir       = "/Users/consti/Desktop/PhD/Publication_material/24) Frost risk/R_output"

#load layers
complete.layers  = stack(paste(raster.frost.dir, "complete.rasters.grd", sep="/"))

#load oceans and land shape files
oceans           = readShapePoly(paste(oceans.dir, "ne_50m_ocean.shp", sep="/"))
land             = readShapePoly(paste(land.dir, "ne_50m_land.shp", sep="/"))



Create map

#creates own color scheme
col1  = colorRampPalette(c("black")) 

#continents and oceans polygons
pols1 = list("sp.polygons", as(oceans, 'SpatialPolygons'), fill = gray(0.99), lwd = 0.1)
pols2 = list("sp.polygons", as(land,   'SpatialPolygons'), fill = gray(0.8), lwd = 0.1)

#spatial points
pts   = list("sp.points", tbl_fread, pch = 19, col = "lightgreen", alpha = 1, cex=0.1)
paste(nrow(pts[[2]]@data),"pixels")
## [1] "4308 pixels"
#colour breaks
brks1 = seq(39,41,0.5)

#get biome map and keep omly temperate and boreal forests/shrublands
map         = complete.layers[["WWF_Biomes_HalfDegree"]]
map[map>8]  = 0
map[map==4] = 40
map[map==5] = 40
map[map==6] = 40
map[map==8] = 40
map[map<39] = NA

FigS4 = spplot(map, 
       maxpixels=2*ncell(complete.layers),
       col.regions = (col1(length(brks1))),
       sp.layout=list(pols1, pols2, pts),
       colorkey =FALSE,
       main=list(label='Fig. S4: GFBi plots',cex=1))

FigS4

#Safe pdf
pdf(paste(output.dir,"FigureS4.pdf",sep="/"), width=12, height=8)
FigS4
